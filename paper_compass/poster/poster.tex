% ---------- Column 1: What a compass is ----------
\begin{column}{0.31\textwidth}

  \begin{block}{1. What is a compass?}
    \begin{itemize}
      \item For head $h$ at layer $\ell$, factor
        $W_{OV} = W_V W_O = U\Sigma V^\top$.
      \item A \textbf{compass plane} is two right-singular vectors
        $(u_i, u_j)$; inject along it:
        $v(\theta, \alpha) = \alpha\sigma_i\cos\theta\,u_i + \alpha\sigma_j\sin\theta\,u_j.$
      \item If the plane is causal, logit-difference swings
        sinusoidally in $\theta$ and linearly in $\alpha$.
    \end{itemize}
  \end{block}

  \begin{block}{2. Sinusoidal response}
    \panelimg{../../helix_usage_validated/phi3_l28h1_causal_polar.png}
    \small
    Phi-3 Mini L28H1 gender compass. Radius = logit($he-she$),
    angle = injection $\theta$. Poles at $0^\circ / 180^\circ$ decode
    to \emph{he / his / him} vs.\ \emph{she / her / hers}.
  \end{block}

  \begin{block}{3. Linearity + phase stability}
    \panelimg{../figures/phi3_linearity.png}
    \small
    Two-pillar test: amplitude $R^2 \geq 0.993$ vs.\ $\alpha$, phase
    drift $\leq 3^\circ$ across $\alpha\in\{1,3,10\}$. Random planes
    and permuted labels kill the signal.
  \end{block}

\end{column}
