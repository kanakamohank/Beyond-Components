% ============================== PANEL 1 ==============================
\paneltitle{1. What is a compass?}{OV-SVD $\rightarrow$ rank-2 dial}

\begin{panelbullets}
\item For head $h$ at layer $\ell$, factor
  $W_{OV} = W_V W_O = U\Sigma V^\top$.
\item A \textbf{compass plane} is two right-singular vectors
  $(u_i, u_j)$. Inject along it:
  \[
    v(\theta, \alpha) \;=\; \alpha\sigma_i\cos\theta\,u_i \;+\;
                              \alpha\sigma_j\sin\theta\,u_j.
  \]
\item If the plane is causal, the logit-difference swings
  \textbf{sinusoidally} in $\theta$ and \textbf{linearly} in $\alpha$;
  if it is just correlational, random planes and label permutations
  reproduce the same numbers.
\item The rest of this deck shows that (i) the sinusoidal +
  linear response really holds, (ii) compasses exist in every model we
  looked at, and (iii) they can drive a debiasing stack.
\end{panelbullets}
\clearpage

\panelbigimg[0.55\textheight]{../figures/phi3_linearity.png}

\begin{panelbullets}
\item Amplitude grows \textbf{linearly} with $\alpha$
  ($R^2 \geq 0.993$).

\item Random planes and permuted labels kill both signals ---
  so the plane is not a geometric accident.

% ============================== PANEL 5 ==============================
\paneltitle{5. Pipeline}{From raw weights to a routed debiasing dial}

\panelbigimg[0.35\textheight]{images/pipeline_schematic.png}

\begin{panelbullets}
\item \textbf{OV-SVD} per head $\rightarrow$
      \textbf{$\alpha$-sweep + 9 null tests} $\rightarrow$
      \textbf{route by pass-count} $\rightarrow$
      \textbf{calibrate} $\alpha_d = \tau / \overline{\mathrm{SNR}}_d$.
\item Blind-scan pass rates vs.\ full-OV null (95\% CI):
      GPT-2 2.06\%, Phi-3 1.18\%, Gemma 2.02\%, Llama 0.22\% ---
      all significantly above the null.
\item Nine null tests knock out correlational planes: permuted labels,
      random planes (1000 MC), full-OV ablate, semantic-ablate,
      principal-angles, causal-patch, self-test, decode-heads,
      cyclicity.
\end{panelbullets}

% ============================== TAKE-HOME ==============================
\paneltitle{Take-home}{Three things to walk away with}

\vspace{1em}
\begin{panelbullets}
\item[\textbf{1.}] Any attention head that writes a categorical axis
  contains a \textcolor{accentA}{\textbf{2D compass plane}} you can
  find by looking at \emph{one SVD}.
\item[\textbf{2.}] The plane is
  \textcolor{accentB}{\textbf{causal, linear, and phase-stable}} ---
  not just a correlation.
\item[\textbf{3.}] Routing + per-domain $\alpha$ turns these planes
  into a \textbf{debiasing dial} that matches or beats INLP /
  SentenceDebias while preserving capability.
\end{panelbullets}
